1. CSS Transition

The following code was used to apply a scale transformation to the images of each project on the home page. When hovering over the image, it scales to increase by 10%. The values for properties such as width, height, border-radius, and scale can be customized.

.picture {

width: 400px;
height: 300px;
border-radius: 20px;
cursor: pointer;
object-fit: cover;
aspect-ratio: 1;
transition: transform 0.5s ease-in-out;

}

.picture:hover {

transform: scale(1.1,1.1);

}

2. JS with p5 library

Mouse Conditions

Mouse Position: mouseY and mouseX can be used to retrieve the exact position of a user's mouse on the canvas. Below is an example of a piece of code that leverages mouseX and mouseY to make the visualization responsive to mouse position.

function setup() {

createCanvas(windowWidth, windowHeight);

}

function draw() {

background(209, 241, 255);
for (let i = 0; i < 20; i++) {

stroke(100 - (mouseX/3), 91 - (8*i), 150 - 2*(mouseY/100));
strokeWeight(1+ mouseX/400);
fill(200 - (mouseX/3), 191 - (8*i), 250 - 2*(mouseY/100));
bezier(mouseY + i*3000, 0 + i*10, mouseX - 4000, mouseY - 1000, 0, 400, 1000, windowHeight + 1000);

}

}

Mouse Pressed Down: The function mousePressed() can be used to specify what happens when a user pressed their mouse down.

Sprites controlled by keyboard keys

The code below sets up a sprite to be controlled by arrow keys. In this example, the sprite is called player.

let player;

window.keyPressed = () => {

if (keyCode == RIGHT_ARROW) {

player.direction = 0;
player.speed = 3.5;

}

else if (keyCode == DOWN_ARROW) {

player.direction = 90;
player.speed = 3.5;

}

else if (keyCode == LEFT_ARROW) {

player.direction = 180;
player.speed = 3.5;

}

else if (keyCode == UP_ARROW) {

player.direction = 270;
player.speed = 3.5;

}

player.direction = 0;
player.speed = 0;

}
return false;

};

Adding multiple sprites to a group

The code below adds multiple sprites to a group and specifies random position and speed for each sprite in the group. In this example, fires is the sprite group and fire is an individual sprite.

let fire;
let fires;

window.setup = () => {

fires = new Group();
for (let i = 0; i < 5; i++) {

fire = new Sprite(random(windowWidth), random(windowHeight)); // (random(windowWidth), random(windowHeight));
fire.img = 'images/fire.png';
fire.speed = random(10);
fire.rotationSpeed = random(10);
fire.position.x = random(windowWidth);
fire.position.y = random(windowHeight);
fires.add(fire); // adds sprite to the fires group
console.log("Length: " + fires.length); // shows how many sprites are in the fires group

}

};
